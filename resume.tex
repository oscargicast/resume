%% --- Custom macros ---------------------------------------------------------
%% Presentation components (reused by content macros and section files).
%% \cvHeading{text}  — bold + titleMid color, used for any sub-title.
%% \cvNote{text}     — free-text line under a \cvEntry (activities, narrative).
%% \cvHonor{t}{d}    — colored title + detail, for honors/certifications items.
\newcommand{\cvHeading}[1]{\textbf{\textcolor{titleMid}{#1}}}
\newcommand{\cvNote}[1]{\noindent #1\par\vspace{2pt}}
\newcommand{\cvHonor}[2]{\cvHeading{#1:} #2}

%% \cvEntry{Title}{Dates}{Org}{Location}
\newcommand{\cvEntry}[4]{%
  \noindent\cvHeading{#1}\hfill #2 \\
  \textit{#3}\hfill \textit{#4}\par\vspace{2pt}%
}

%% \cvProject{Name}{Stack}{URL without scheme, e.g. "github.com/foo/bar"}
%% Empty third arg = no URL. Visible text stays short (no https://); the
%% clickable target gets https:// prepended.
\newcommand{\cvProject}[3]{%
  \noindent\cvHeading{#1} \textbar\ \textit{#2}%
  \ifx&#3&\else\hfill\href{https://#3}{#3}\fi
  \par\vspace{2pt}%
}

%% \cvSkill{Category}{Comma-separated list}
\newcommand{\cvSkill}[2]{%
  \noindent\cvHeading{#1:} #2\par\vspace{1pt}%
}
